Examples

class DefaultBuild : NukeBuild
{
    public static int Main () => Execute<DefaultBuild>(x => x.Compile);

    Target Clean => _ => _
        .Executes(() =>
        {
            EnsureCleanDirectory(OutputDirectory);
        });

    Target Compile => _ => _
        .DependsOn(Clean)
        .Executes(() =>
        {
            MSBuild(SolutionFile);
        });
}
